  <script>

    const menuBtn = document.getElementById('menuButton');

    const overlay = document.getElementById('menuOverlay');

    const closeBtn = document.getElementById('closeOverlay');


    function setOverlay(open){

      overlay.setAttribute('data-open', String(open));

      menuBtn.setAttribute('aria-expanded', String(open));

      document.body.style.overflow = open ? 'hidden' : '';

    }


    menuBtn.addEventListener('click', () => setOverlay(true));

    closeBtn.addEventListener('click', () => setOverlay(false));


    // Click outside inner panel closes

    overlay.addEventListener('click', (e)=>{ if(e.target===overlay) setOverlay(false); });


    // Keyboard handling

    document.addEventListener('keydown', (e)=>{ if(e.key==='Escape') setOverlay(false); });


    // Accordion behavior for tabs

    document.querySelectorAll('.tab').forEach(tabBtn => {

      tabBtn.addEventListener('click', () => {

        const controls = tabBtn.getAttribute('aria-controls');

        const panel = document.getElementById(controls);

        const expanded = tabBtn.getAttribute('aria-expanded') === 'true';

        // close others

        document.querySelectorAll('.tab').forEach(b => { if (b!==tabBtn) b.setAttribute('aria-expanded','false'); });

        document.querySelectorAll('.submenu').forEach(s => { if (s!==panel) s.setAttribute('data-open','false'); });

        tabBtn.setAttribute('aria-expanded', String(!expanded));

        panel.setAttribute('data-open', String(!expanded));

      });

    });

  </script>
